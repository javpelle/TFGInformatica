operation MainQuantum() : String {
	//Select desired Qubit number to be used
	using (register = Qubit[5]) {
            
            
		//Call method
		let s = DeutschJozsa(register, uf);

		
		//Reset all qubits to Zero state

		return s;	
	    }
    }

        CNOT(register[0], register[Length(register) - 1]);
